Можно отделить аналитику от ViewModel и от всех Сервисов, если надо. 

Есть 2 способа:

 Первый
 ```swift
let analytics = Analytics()
let service = ItemService()
service.onError = analytics.track(Event)
```

Второй, если сервис не выдает `onError`

```swift
class ItemServiceTrackingDecorator: ItemService
	private let analytics = Analytics ()
	private let decoratee: ItemService
	private let key: String
	
	init (decoratee: ItemService, key: String) {
		self.decoratee = decoratee
		self.key = key
	}
	
	func load(itemWithID id: String,
	 completion: @escaping (ItemService.Result) → Void) {
		decoratee.load(itemWithID: id) { [analytics, key] result in
			completion (result.mapError {
				analytics.track(key: key, value: id)
				return $0
				} 
			)
		}
	}
}
```
