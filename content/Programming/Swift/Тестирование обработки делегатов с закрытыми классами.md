```swift
 protocol AppleIDCredential {
    var identityToken: Data? { get }
    var user: String { get }

    var fullName: PersonNameComponents { get }

}

extension AuthAppleIDController: ASAuthorizationControllerDelegate {

    func didCompleteWith(credentials: AppleIDCredential) {
        // Create own needed protocol or params and send in new method 
		// Some logic
    }

    func authorizationController(
	    controller: ASAuthorizationController,
		didCompleteWithAuthorization authorization: ASAuthorization
	) {

// How test logic in this block? `ASAuthorization` has closed init.
        guard let credential = authorization.credential as? AppleIDCredential else { return }

        didCompleteWith(credentials: credential)
    }
}
```
